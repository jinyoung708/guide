세로 막대 그래프

                                <canvas id="xBarChart"></canvas>
                            
                                var xBarChartEl = document.getElementById('xBarChart').getContext('2d');
var xBarChart = new Chart(xBarChartEl, {
    type: 'bar',
    data: {
        labels: ['GRADE1', 'GRADE2', 'ETC'],
        datasets: [{
            data: [15, 60, 30],
            backgroundColor: ['#ff6384', '#36a2eb', '#ff9f40'],
            borderColor: ['#ff6384', '#36a2eb', '#ff9f40'],
            borderWidth: 1
        }, ],
    },
    plugins: [ChartDataLabels],
    options: {
        plugins: {
            datalabels: {
                color: '#000000',
                clip: true,
                anchor: 'end',
                align: 'start',
                clamp: true,
                offset: 10
            },
            legend: {
                display: false
            },
        },
        scales: {
            x: {
                display: false
            },
            y: {
                display: false
            }
        }
    }
});

가로 적층형 막대 그래프

                                <canvas id="yBarChart"></canvas>
                            
                                var yBarChartEl = document.getElementById('yBarChart').getContext('2d');
var yBarChart = new Chart(yBarChartEl, {
    type: 'bar',
    data: {
        labels: ["찬반"],
        datasets: [{
            label: "찬성",
            data: [40],
            backgroundColor: ['#36a2eb']
        }, {
            label: "반대",
            data: [60],
            backgroundColor: ['#ff6384']
        }]
    },
    plugins: [ChartDataLabels],
    options: {
        maintainAspectRatio: false,
        indexAxis: 'y',
        scales: {
            x: {
                stacked: true,
                display: false 
            },
            y: {
                stacked: true,
                display: false
            }
        },
        plugins: {
            legend: {
                display: false
            },
            datalabels: {
                color: '#000000'
            }
        }
    }
});

도넛 차트

                                <canvas id="doughnutChart"></canvas>
                            
                                var doughnutChartEl = document.getElementById('doughnutChart').getContext('2d');
var doughnutChart = new Chart(doughnutChartEl, {
    type: 'doughnut',
    data: {
        labels: ['GRADE1', 'GRADE2', 'ETC'],
        datasets: [{
            data: [33, 35, 20],
            backgroundColor: ['#ff6384', '#36a2eb', '#ff9f40'],
            borderColor: ['#ff6384', '#36a2eb', '#ff9f40'],
            borderWidth: 1
        }, ],
    },
    plugins: [ChartDataLabels],
    options: {
        animation: {
            duration: 1000
        },
        plugins: {
            datalabels: {
                color: '#ffffff',
                clip: true,
                anchor: 'end',
                align: 'end',
                offset: 4
            },
            legend: {
                onClick: function (e) {
                    e.stopPropagation()
                }
            }
        }
    }
});
